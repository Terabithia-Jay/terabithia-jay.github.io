                      colorList: [...this.data.colorList, ...res.data]
                  })
              },
              
              // 隐藏loading提示框
              complete: () => {
                 wx.hideLoading() 
              }
          })
      },

WXS​

  • wxml中无法调用在页面的.js中定义的函数(指的是不能在插值语法调用函数) ，但是，wxml中可以调用wxs中定义的函数。因此，小程序中wxs的典型应用场景就是“过滤器”。

  • 不能作为组件的事件回调

    <button bindtap="m2.toLower">按钮</button>

  • 不能调用js中定义的函数

  • 不能调用小程序提供的API函数

内联wxs脚本​

// .wxml
<view>{{m1.toUpper(username)}}</view>

<wxs module='m1'>
    // 将文本转为大写形式 zs -> ZS
    module.exports.toUpper = function(str){
        return str.toUpperCase()
    }
</wxs>
// .js
data: {
    username: 'zs'  
}

外联wxs脚本​

  • wxs代码还可以编写在以.wxs为后缀名的文件内
// tools.wxs：
function toLower(str){
    return str.toLowerCase()
}

module.exports = {
    toLower: toLoWer
}
  • module用来指定模块的名称
  • src用来指定要引入的脚本的路径，且必须是相对路径
// .js
data: {
    country: 'CHINA'  
}
// .wxml
<!-- 调用 m2 模块中的方法 -->

<!-- 引入外联的 tools.wxs 脚本，并命名为 m2 -->

动态加载第几页数据​

// .json
{
    "usingComponents": {},
    "onReachBottomDistance": 200,
    "enablePullDownRefresh": true,
    "backgroundColor": "#efefef",
    "backgroundTextStyle": "dark"
}
data: {
    // 点击传过来的数据（id和title）
    query: {},
    //商品列表数据
    shopList: [],
    // 请求第几页的数据
    page: 1,
    // 请求pagesize
    pageSize: 10,
    // 拿到数据的总数
    total: 0,
    // 节流阀
    isLoading: false
},
    
onLoad(options) {
    this.setData({
        query: options
    })
    this.getShopList()
},

getShopList(cb) {
    // 节流，正在发送请求
    this.setData({
        isLoading: true
    })
    // loading
    wx.showLoading({
        title: '数据加载中...',
    })

    wx.request({
        url: `https://www.escook.cn/categories/${this.data.query.id}/shops`,
        method: "GET",
        data: {
            // 请求第几页的数据
            _page: this.data.page,
            // 请求一页的大小
            _limit: this.data.pageSize
        },
        success: (res) => {
            this.setData({
                shopList: [...this.data.shopList, ...res.data],
                total: res.header['X-Total-Count'] - 0 // 转为数字类型
            })
        },
        // 隐藏loading
        complete: () => {
            wx.hideLoading()
            // 节流阀设置为false
            this.setData({
                isLoading: false
            })
            // 下拉刷新自动合上，但是这个只需要用在下拉刷新的时候才需要调用这个函数，写在这里每次其他地方调用getShopList都会触发这个函数，显然要按需调用才行
            // wx.stopPullDownRefresh()
            cb && cb()
        }
    })
},

onReachBottom() {
    // 判断请求的数据是否超过了总的数据条数，超过则不再发起请求
    if (this.data.page * this.data.pageSize >= this.data.total) {
        // 没有下一页了，提示框
        return wx.showToast({
            title: '数据加载完毕！',
        })
    }
    
    // 节流，如果正在发起请求，则return
    if (this.data.isLoading) return
    
    // 下拉加载更多数据
    this.setData({
        page: this.data.Page + 1
    })
    this.getShopList()
},
    
onPullDownRefresh() {
    // 下拉刷新的时候加载第一页的数据
    this.setData({
        page: 1,
        shopList: [],
        total:0
    })
    // 传一个回调函数过去按需调用下拉刷新回弹
    this.getShopList(() => {
        wx.stopPullDownRefresh()
    })
},

组件的监听​

Component({
    data: { 
        n1: 0, 
        n2: 0, 
        sum: 0,
        rgb:{
            r: 0,
            g: 0,
            b: 0
        },
        fullColor: ''
    },
    method: {
        addN1(){
            this.setData({
                n1: this.data.n1 + 1 
            })
        },
        addN2(){
            this.setData({ 
                n2: this.data.n2 + 1 
            })
        }
    },
    observers: {
        // 监听简单数据
        'n1, n2': funcion(newN1, newN2){
            this.setData({ 
                sum: newN1 + newN2 
            })
        },
         // 监听对象中的属性
         'rgb.r, rgb.g, rgb.b': function(newR, newG, newB){
            this.setData({
                fullColor: `${r}, ${g}, ${b}`
            })
        }
        // 上面监听多个属性可以这么写：
        'rgb.**': function(newObj){
            this.setData({
                fullColor: `${newObj.r}, ${newObj.g}, ${newObj.b}`
            })
         }
    }
})

状态管理​

  • 创建store实例对象：mobx-miniprogram
  • 把store的共享数据或方法绑定到组件或页面中使用：mobx-miniprogram-bindings

  1. 安装：npm install --save mobx-miniprogram@4.13.2 mobx-miniprogram-bindings@1.2.1

  2. 构建npm

  3. 目录下创建store文件夹，store.js文件

    import { observable, action } from 'mobx-miniprogram'
    export const store = observable({
        // 数据
        numA: 1,
        numB: 2,
        
        // 计算属性
        get sum() {
            return this.numA + this.numB
        }
        
        // actions用来修改store中的数据
        updataNum1: action(function(step) {
            this.numA += step
        })
        updataNum2: action(function(step) {
            this.numB += step
        })
    })

  4. 引入store

    • 在页面中引入

      import { createStoreBindings } from 'mobx-miniprogram-bindings'
      import { store } from '../../store/store'

      Page({
          onLoad: function() {
              this.storeBindings = createStoreBindings(this, {
                  store,
                  // 要绑定的数据字段，这些字段可以直接在wxml直接使用，不用加this
                  fields: ['numA','numB','sum'],
                  // 要绑定的方法
                  actions: ['updateNum1']
              })
          },
          
          onUnload: function() {
              // 卸载销毁
              this.storeBindings.destoryStoreBindings()
          }
      })

    • 在组件中引入

      import { storeBindingsBehavior } from 'mobx-miniprogram-bindings'
      import { store } from '../../store/store'

      Component({
          behaviors: [storeBindingsBehavior]
          
          storeBindings: {
              store,
              // 要绑定的数据字段
              fields: {
                  // 绑定字段的三种方式
                  numA: () => store.numA,
                  numB: (store) => store.numB,
                   sum: "sum"
              }
              // 要绑定的方法
              actions: {
                  updateNum2: updateNum1
              }
          },
          
          onUnload: function() {
              // 卸载销毁
              this.storeBindings.destoryStoreBindings()
          }
      })

  5. 使用store

    • 页面使用

      // .wxml
      <view>{{numA}} + {{numB}} = {{sum}}</view>  // 可以直接使用js中fields的字段

      <van-button type="primary" bindtap="btnHandler1" data-step="{{1}}">     // data-xxx 向点击事件传参
          numA + 1
      </van-button>

      <van-button type="danger" bindtap="btnHandler1" data-step="{{-1}}">
          numA - 1
      </van-button>

      // .js
      btnHandler1(e) {
          this.updateNum1(e.target.dataset.step)
      }

    • 组件使用

      // .wxml
      <view>{{numA}} + {{numB}} = {{sum}}</view>

      <van-button type="primary" bindtap="btnHandler2" data-step="{{1}}">     // data-xxx 向点击事件传参
          numA + 1
      </van-button>

      <van-button type="danger" bindtap="btnHandler2" data-step="{{-1}}">
          numA - 1
      </van-button>

      // .js
      methods: {
          btnHandler2(e) {
              this.updateNum1(e.target.dataset.step)
          }
      }

分包​

  • 访问默认会把全部的页面加载，分包就是按需加载页面，首次只加载首页
  • 小程序启动时会默认下载主包并启动主包内的页面（tabBar要放到主包中）
  • 进入分包某个页面时，客户端才会把对应的分包下载下来
  • 非tabBar页面可以按照功能的不同划分为不同的分包后，进行按需下载
  • 整个小程序所有的分包+主包不能超过20M，单个分包不能超过2M
  • 主包无法引用分包和独立分包内的私有资源，独立分包间、独立分包和普通分包间不能互相引用私有资源。独立分包不能引用主包内的公共资源
  • 独立分包：默认访问只能从主包，然后进入分包。如果想从其他地方访问不通过主包（不用加载主包），就需要独立分包

配置普通分包​

// app.js 
"subPackages": [
    {
        "root": "pkgA",     // 根目录名
        "name": "p1"        // 别名
        "pages": [
           "pages/cat/cat",     // 包文件名
            "pages/dog/dog",
        ]
    },
    {
        "root": "pkgB",
        "name": "p2"
        "pages": [
            "pages/apple/apple",
            "pages/dog/dog",
        ]
    }
],

配置独立分包​

// app.js 
"subPackages": [
    {
        "root": "pkgA",     // 普通分包
        "name": "p1"        
        "pages": [
           "pages/cat/cat",     
            "pages/dog/dog",
        ]
    },
    {
        "root": "pkgB",     // 独立分包
        "name": "p2"
        "pages": [
            "pages/apple/apple",
            "pages/dog/dog",
        ],
        "independent": true     // 设置为独立分包
    }
],

分包预下载​

  • 进入指定页面后就加载某个分包（所有的预下载分包加起来最大限额2M），从而提升后续访问这个分包的加载速度
// app.js
{
    "perloadRule": {
        "pages/contact/contact": {      // 触发分包的页面路径，访问这个页面就事先加载 pkgA 这个分包
            "network": "all",       // all不限网络（或可写wifi，只限wifi）
            "packages": ['pkgA']    // 这里数组可以写root或者name的值
        }
    }
}
