1.通过script标签使用React​

  • 下载：

    • https://unpkg.com/react@18.0.0/umd/react.development.js

    • https://unpkg.com/react-dom@18.0.0/umd/react-dom.development.js

  • 引入：

    <head>
        <meta charset="UTF-8">
        <title>Hello World</title>
        <!--引入react的核心库-->
        <script src="script/react.development.js"></script>
        <!--引入react的DOM库-->
        <script src="script/react-dom.development.js"></script>
    </head>
    <body>
        <div id="root"></div>
        <script>
            // 原生js通过DOM向页面中添加一个div：
                // 创建一个div
                    const div = document.createElement('div'); // DOM元素
                // 向div中设置内容
                    div.innerText = '我是一个div';
                // 获取root
                    const root = document.getElementById('root');
                // 将div添加到页面中
                    root.appendChild(div);


            // React向页面中添加一个div：
                /*
                   一、React.createElement()  创建一个React元素
                       - 参数：
                           1. 元素名、组件名（如果是元素必须全是小写，大写的是组件名）
                           2. 元素中的属性
                                - id，class用className
                                - 如果是绑定事件必须得是驼峰命名，传入的必须是函数：onClick: () => { alert(123) } 
                           3. 元素的子元素（内容）
                   二、ReactDOM.createRoot()  创建React根元素
                   三、root.render()          渲染
                */
                // 1、创建一个React元素，这里用的react核心库
                   const div = React.createElement('div', {
                        id: "hello",
                        className: "hello",
                        type: "button",
                        onClick: () => { alert(123) }
                    }, '我是React创建的div', button); // React元素

                // 获取根元素（容器root）对应的React元素
                // 2、创建React根元素，需要一个DOM元素作为参数，这里有使用dom，把dom元素转为react元素，用的react的dom库
                    const root = ReactDOM.createRoot(document.getElementById('root'));

                // 3、将div渲染添加到根元素中（render）
                    root.render(div);
        </script>
    </body>

  • 注意：

    • dom元素不能与React元素一起混用（类似于原生js和jquery，要混用必须得转成一样的）
    • React元素一旦创建createElement就不能修改，只能通过新创建的元素然后渲染进行替换
      • 修改替换元素后，必须重新渲染（render），渲染时，React会自动比较两次渲染的元素（react的虚拟dom和js的真实dom），只在真实dom中更新发生变化的部分，先比较父元素，如果父元素不同，所有元素全部替换，如果父元素一样，则逐个对比子元素，按顺序从上到下对比（diffing算法）

JSX（JS扩展）​

  • 声明式编程，结果导向的编程

  • JSX就是 React.createElement() 的语法糖

  • 通过JSX创建React元素，需要翻译（用babel把JSX的代码转换为React代码）

    1. 下载：https://unpkg.com/babel-standalone@6/babel.min.js

    2. 引入

      <script src="script/babel.min.js"></script>

    3. 设置js代码被babel处理：

      <body>
          <div id="root"></div>

          <!--设置js代码被babel处理-->
          <script type="text/babel">
              // 在这里面就可以直接写html代码即可，不需要再React.createElement()
              // 标签小写开头，组件大写。有且只能有一个根标签，最大的div包裹里面
              // 属性可以直接在标签中设置
              const div = <div className="box" onClick={()=>{alert("haha")}} style={{backgroundColor:'red'}}> 
                  我是一个div
                  <button>我是按钮</button>

                  const name = '孙悟空'
                  <div>
                      {name}  // 使用{}获取上面的变量name
                  </div>
              </div>;

              const root = ReactDOM.createRoot(document.getElementById('root'));

              root.render(div);
          </script>
      </body>
      • 注意：
        • 标签小写开头，组件大写
        • 有且只能有一个根标签，最大的div包裹里面
        • 标签必须正确结束（自结束标签必须写/）
        • 可以使用 { } 嵌入表达式（有值的语句）
          • 如果表达式是空值、布尔值、undefined，将不会显示
        • 属性可以直接在标签中设置
          • class需要使用className代替
          • 事件用驼峰命名法
          • style中必须使用对象 { } 设置
            • style={{backgroundColor:'red'}}

key​

  • 渲染对比是从上往下进行的，如果在前面插入新的元素，但此时对比的位置发生的改变，修改的元素发生了错位

  • 解决办法：给一个key属性（跟vue一样）

  • 注意：

    1. 开发中一般会采用数据的id作为key（必须唯一的）
    2. 尽量不要使用元素的index作为key（除非顺序不发生变化才可以使用）
    // 可以在jsx中直接写document.getElementById('btn').onclick = function (){}
    document.getElementById('btn').onclick = function (){
        const data = ['唐僧', '孙悟空', '猪八戒', '沙和尚'];
        // 创建列表
        const list = <ul>
              // {data.map(item => <li>{item}</li>)}

              // 添加key属性
              {data.map(item => <li key={id}>{item}</li>)}
              {data.map((item, index) => <li key={index}>{item}</li>)}
        </ul>;
        // 渲染元素
        root.render(list);
    }

2.手动创建React项目​

  1. 目录下创建public和src文件夹

    1. public下创建index.html文件

      • index.html是首页的模板，webpack在编译打包文件时，会以index.html为模板在build文件夹下生成index.html
      <!DOCTYPE html>
      <html lang="en">
      <head>
          <meta charset="UTF-8">
          <meta http-equiv="X-UA-Compatible" content="IE=edge">
          <meta name="viewport" content="width=device-width, initial-scale=1.0">
          <title>Document</title>
      </head>
      <body>
          <div id="root"></div>   // 创建根
      </body>
      </html>

    2. src下创建index.js文件

      // 引入ReactDOM
          import ReactDOM from 'react-dom/client';    // 引入文件后面要加/client

      // 创建一个JSX
          const App = <div>
              <h1>这是一个React项目</h1>
              <p>我终于有了第一个React项目了!!!</p>
          </div>;

      // 获取一个根容器
          const root = ReactDOM.createRoot(document.getElementById('root'));

      // 将App渲染进根容器
          root.render(App);

  2. 初始化项目并安装依赖

    yarn init -y

    yarn add react react-dom react-scripts

  3. package.json文件设置快捷键

    "scripts": {
        "start": "react-scripts start",
        "build": "react-scripts build"
      },

  4. 打开内置服务器预览页面

    yarn start

  5. 构建打包

    yarn build

  6. 把build文件夹下的index.html中的路径改为相对路径

    src="./static/js/main.593b4e52.js"

类组件​

  • 类组件跟函数组件功能是一样的
// App.js：
import React from "react";

class App extends React.Component{
    // 类组件中，必须添加一个render()方法，且方法的返回值要是一个jsx

        return <div>我是一个类组件</div>;
    }
}

export default App;

// index.js：
import ReactDOM from "react-dom/client";
import App from "./App";

root.render(<App/>);

事件​

  • 事件需要通过元素的属性来设置，使用驼峰命名法

    • onclick -> onClick
    • onchange -> onChange

  • 属性值不能直接执行代码，而是需要一个回调函数：

    onclick="alert(123)"    // 原生

    onClick={() => {    // react
        alert(123)
    }}

    <button onClick={() => {
            alert(123);
        }}>点我一下
    </button>

    // 事件对象event：
    const clickHandler = (event) => {
        alert('我是App中的clickHandler！');
        return false;   // 原生js在函数最后return false取消默认行为（页面跳转）
        // 在React中，无法通过return false取消默认行为
        
        // 事件对象event：
            // React中的事件event对象不是原生的事件对象，是经过React包装后的事件对象（虚拟dom）
        event.preventDefault(); // react取消默认行为
        event.stopPropagation(); // react取消事件的冒泡
    };
    <button onClick={clickHandler}>哈哈</button>      // clickHandler不能加括号
    <a href="https://www.baidu.com" onClick={clickHandler}>超链接</a>

state（响应式数据）​

  • 原因：修改变量不是响应式的，不会重新渲染

    • state和props类似，都是一种存储属性的方式，但是不同点在于state只属于当前组件，其他组件无法访问，只能自己用
    • state是可变的，当其发生变化后组件会自动重新渲染，以使变化在页面中呈现

  • 在函数组件中，我们需要通过钩子函数获取state

    • 它需要一个值作为参数，这个值就是state的初始值
    • 该函数会返回一个数组
      • 第一个元素：初始值（写入的什么就是什么）
        • 获取：let counter = useState(1)[0]
        • 初始值只用来显示数据在页面，直接修改不会触发组件的数据重新渲染（更改了数据不会响应式）
      • 第二个元素：函数，通常会命名为setXxx
        • 获取：let setCounter = useState(1)[1]
        • 这个函数用来修改数据state，调用其修改数据state后会触发组件的重新渲染，并且使用函数的值作为新的state值（在函数setCounter里修改数据才是响应式的）
        • 如果函数实参与原本的初始值一致，那仍不会重新渲染（diff）
    import {useState} from "react";

    const App = () => {
      // let result = useState(1)
      // let counter = result[0];
      // let setCounter = result[1];
      // const [counter, setCounter] = result;
      
      // 使用 useState()
      const [counter, setCounter] = useState(1);    // 解构

      const addHandler = () => {
        // counter++        直接在这里修改counter不行，修改了但没重新渲染
        setCounter(counter + 1);    // 更改counter的值，将counter值++
      };

      const lessHandler = () => {
        setCounter(counter - 1);
      };

      return <div className={'app'}>
        <h1>{counter}</h1>
        <button onClick={lessHandler}>-</button>
        <button onClick={addHandler}>+</button>
      </div>;
    };
    export default App;

  • 注意：

    • 当state的值是一个对象时，修改时是使用新的对象去替换覆盖已有对象

      • 对象判断有没有修改比较的是内存地址而不是对象里面的属性和值，即使新旧对象一模一样也会重新渲染，因为地址不一样。如果新旧对象的地址一样，属性和值不一样，也不会重新渲染
      • 当通过setState去修改一个state（第一个参数）时，并不表示修改了第一个参数的值（初始值一开始是多少就是多少，不会改），它修改的是组件下一次渲染时state的值
      const [user, setUser] = useState({name: '孙悟空', age: 18});   // user就是{name: '孙悟空', age: 18}
      const updateUserHandler = () => {
          setUser({name:'猪八戒'});      // 把{name:'猪八戒'}替换{name: '孙悟空', age: 18}
          
          // 如果只想修改对象中某一个属性：
              // 直接修改旧的state对象，由于对象还是那个对象，没有变，所以不会生效
                  // user.name = '猪八戒';   
                      // 这里直接修改了user里原来的那个对象{name: '孙悟空', age: 18}，即使修改成功了，但是react进行对比的时候，对象进行对比的时候比的是地址，而不是对象里面的属性和值的不同，地址没变，所以不会重新渲染
                  // console.log(user);
                  // setUser(user);
          
              // 1、浅拷贝
                 const newUser = Object.assign({}, user);     // assign() 将use复制一份给{}
                      // console.log(newUser === user)  -->   false，长得一样，但是地址不同了
                  newUser.name = '猪八戒';   // 修改里面的值
                  setUser(newUser);       // 将新的对象覆盖旧的{name: '孙悟空', age: 18}对象
              // 2、浅拷贝展开
                  setUser({...user, name: '猪八戒'});    // 在对象{}中把对象user展开，相当于是浅拷贝给这个对象
      };
      <button onClick={updateUserHandler}>点击</button>

    • setState()会触发组件的重新渲染，它是异步的（修改state的值不会立刻渲染：index.js中的root.render(

      )）

      import {useState} from "react"
      import "./App.css"
      export default function App() {
          const [counter, setCounter] = useState(0)
          console.log("我被渲染了！")
          function changeCounter(){
              setCounter(1)       // 队列1（先进先出）
              setCounter(2)       // 队列2
              setCounter(3)       // 队列3
              setCounter(4)       // 队列4
              setCounter(5)       // 执行覆盖前面所有队列，执行渲染一次，而不是调用一次setCount就会渲染一次
          }
          return <div className="app">
              <h1>{counter}</h1>
              <button onClick={changeCounter}>多次修改</button>
          </div>
      }

      • 所以当调用setState()需要用旧的state的值时，一定要注意有可能出现计算错误的情况（两次快速点击拿到的值都是旧的队列中上一个state的值，两次拿到的值都是上一次队列的state的值，值相同，所以点击了两次实则只改了一次）

        • 为了避免这种情况，可以通过为setState()传递回调函数的形式来修改state值

          import {useState} from "react"
          import "./App.css"
          export default function App() {
              let[counter, setCounter] = useState(0)
              function changeCounter() {
                setTimeout(() => {
                    // setCounter(counter++)

                    // setState()中回调函数的返回值将会成为新的state值，回调函数执行时，React会将最新的state值作为参数传递
                      // setCounter((prevCounter) => {        取一个名字prevCounter形参，接收state的值
                      //     return prevCounter + 1;
                      // });
                      setCounter(preCount => preCount + 1)
                }, 1000)
              }
              return <div className="app">
                 <h1>{counter}</h1>
                 <button onClick={changeCounter}>点击</button>
              </div>
          }
