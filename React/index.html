1.通过script标签使用React​

  • 下载：

    • https://unpkg.com/react@18.0.0/umd/react.development.js

    • https://unpkg.com/react-dom@18.0.0/umd/react-dom.development.js

  • 引入：

    <head>
        <meta charset="UTF-8">
        <title>Hello World</title>
        <!--引入react的核心库-->
        <script src="script/react.development.js"></script>
        <!--引入react的DOM库-->
        <script src="script/react-dom.development.js"></script>
    </head>
    <body>
        <div id="root"></div>
        <script>
            // 原生js通过DOM向页面中添加一个div：
                // 创建一个div
                    const div = document.createElement('div'); // DOM元素
                // 向div中设置内容
                    div.innerText = '我是一个div';
                // 获取root
                    const root = document.getElementById('root');
                // 将div添加到页面中
                    root.appendChild(div);


            // React向页面中添加一个div：
                /*
                   一、React.createElement()  创建一个React元素
                       - 参数：
                           1. 元素名、组件名（如果是元素必须全是小写，大写的是组件名）
                           2. 元素中的属性
                                - id，class用className
                                - 如果是绑定事件必须得是驼峰命名，传入的必须是函数：onClick: () => { alert(123) } 
                           3. 元素的子元素（内容）
                   二、ReactDOM.createRoot()  创建React根元素
                   三、root.render()          渲染
                */
                // 1、创建一个React元素，这里用的react核心库
                   const div = React.createElement('div', {
                        id: "hello",
                        className: "hello",
                        type: "button",
                        onClick: () => { alert(123) }
                    }, '我是React创建的div', button); // React元素

                // 获取根元素（容器root）对应的React元素
                // 2、创建React根元素，需要一个DOM元素作为参数，这里有使用dom，把dom元素转为react元素，用的react的dom库
                    const root = ReactDOM.createRoot(document.getElementById('root'));

                // 3、将div渲染添加到根元素中（render）
                    root.render(div);
        </script>
    </body>

  • 注意：

    • dom元素不能与React元素一起混用（类似于原生js和jquery，要混用必须得转成一样的）
    • React元素一旦创建createElement就不能修改，只能通过新创建的元素然后渲染进行替换
      • 修改替换元素后，必须重新渲染（render），渲染时，React会自动比较两次渲染的元素（react的虚拟dom和js的真实dom），只在真实dom中更新发生变化的部分，先比较父元素，如果父元素不同，所有元素全部替换，如果父元素一样，则逐个对比子元素，按顺序从上到下对比（diffing算法）

JSX​

  • 声明式编程，结果导向的编程

  • JSX就是 React.createElement() 的语法糖

  • 通过JSX创建React元素，需要翻译（用babel把JSX的代码转换为React代码）

    1. 下载：https://unpkg.com/babel-standalone@6/babel.min.js

    2. 引入

      <script src="script/babel.min.js"></script>

    3. 设置js代码被babel处理：

      <body>
          <div id="root"></div>

          <!--设置js代码被babel处理-->
          <script type="text/babel">
              // 在这里面就可以直接写html代码即可，不需要再React.createElement()
              // 标签小写开头，组件大写。有且只能有一个根标签，最大的div包裹里面
              // 属性可以直接在标签中设置
              const div = <div className="box" onClick={()=>{alert("haha")}} style={{backgroundColor:'red'}}> 
                  我是一个div
                  <button>我是按钮</button>

                  const name = '孙悟空'
                  <div>
                      {name}  // 使用{}获取上面的变量name
                  </div>
              </div>;

              const root = ReactDOM.createRoot(document.getElementById('root'));

              root.render(div);
          </script>
      </body>
      • 注意：
        • 标签小写开头，组件大写
        • 有且只能有一个根标签，最大的div包裹里面
        • 标签必须正确结束（自结束标签必须写/）
        • 可以使用 { } 嵌入表达式（有值的语句）
          • 如果表达式是空值、布尔值、undefined，将不会显示
        • 属性可以直接在标签中设置
          • class需要使用className代替
          • 事件用驼峰命名法
          • style中必须使用对象 { } 设置
            • style={{backgroundColor:'red'}}

key​

  • 渲染对比是从上往下进行的，如果在前面插入新的元素，但此时对比的位置发生的改变，修改的元素发生了错位

  • 解决办法：给一个key属性（跟vue一样）

  • 注意：

    1. 开发中一般会采用数据的id作为key（必须唯一的）
    2. 尽量不要使用元素的index作为key（除非顺序不发生变化才可以使用）
    // 可以在jsx中直接写document.getElementById('btn').onclick = function (){}
    document.getElementById('btn').onclick = function (){
        const data = ['唐僧', '孙悟空', '猪八戒', '沙和尚'];
        // 创建列表
        const list = <ul>
              // {data.map(item => <li>{item}</li>)}

              // 添加key属性
              {data.map(item => <li key={id}>{item}</li>)}
              {data.map((item, index) => <li key={index}>{item}</li>)}
        </ul>;
        // 渲染元素
        root.render(list);
    }

  1. 目录下创建public和src文件夹

    1. public下创建index.html文件

      • index.html是首页的模板，webpack在编译打包文件时，会以index.html为模板在build文件夹下生成index.html
      <!DOCTYPE html>
      <html lang="en">
      <head>
          <meta charset="UTF-8">
          <meta http-equiv="X-UA-Compatible" content="IE=edge">
          <meta name="viewport" content="width=device-width, initial-scale=1.0">
          <title>Document</title>
      </head>
      <body>
          <div id="root"></div>   // 创建根
      </body>
      </html>

    2. src下创建index.js文件

      // 引入ReactDOM
          import ReactDOM from 'react-dom/client';    // 引入文件后面要加/client

      // 创建一个JSX
          const App = <div>
              <h1>这是一个React项目</h1>
              <p>我终于有了第一个React项目了!!!</p>
          </div>;

      // 获取一个根容器
          const root = ReactDOM.createRoot(document.getElementById('root'));

      // 将App渲染进根容器

  2. 初始化项目并安装依赖

    yarn init -y

    yarn add react react-dom react-scripts

  3. package.json文件设置快捷键

    "scripts": {
        "start": "react-scripts start",
        "build": "react-scripts build"
      },

  4. 打开内置服务器预览页面

    yarn start

  5. 构建打包

    yarn build

  6. 把build文件夹下的index.html中的路径改为相对路径

    src="./static/js/main.593b4e52.js"

事件​

  • 事件需要通过元素的属性来设置，使用驼峰命名法

    • onclick -> onClick
    • onchange -> onChange

  • 属性值不能直接执行代码，而是需要一个回调函数：

    onclick="alert(123)"    // 原生

    onClick={() => {    // react
        alert(123)
    }}

    <button onClick={() => {
            alert(123);
        }}>点我一下
    </button>

    // 事件对象event：
    const clickHandler = (event) => {
        alert('我是App中的clickHandler！');
        return false;   // 原生js在函数最后return false取消默认行为（页面跳转）
        // 在React中，无法通过return false取消默认行为
        
        // 事件对象event：
            // React中的事件event对象不是原生的事件对象，是经过React包装后的事件对象（虚拟dom）
        event.preventDefault(); // react取消默认行为
        event.stopPropagation(); // react取消事件的冒泡
    };
    <button onClick={clickHandler}>哈哈</button>      // clickHandler不能加括号
    <a href="https://www.baidu.com" onClick={clickHandler}>超链接</a>

state（响应式数据）​

  • 原因：修改变量不是响应式的，不会重新渲染

    • state和props类似，都是一种存储属性的方式，但是不同点在于state只属于当前组件，其他组件无法访问，只能自己用
    • state是可变的，当其发生变化后组件会自动重新渲染，以使变化在页面中呈现

  • 在函数组件中，我们需要通过钩子函数获取state

    • 它需要一个值作为参数，这个值就是state的初始值
    • 该函数会返回一个数组
      • 第一个元素：初始值（写入的什么就是什么）
        • 获取：let counter = useState(1)[0]
        • 初始值只用来显示数据在页面，直接修改不会触发组件的数据重新渲染（更改了数据不会响应式）
      • 第二个元素：函数，通常会命名为setXxx
        • 获取：let setCounter = useState(1)[1]
        • 这个函数用来修改数据state，调用其修改数据state后会触发组件的重新渲染，并且使用函数的值作为新的state值（在函数setCounter里修改数据才是响应式的）
        • 如果函数实参与原本的初始值一致，那仍不会重新渲染（diff）
    import {useState} from "react";

    const App = () => {
      // let result = useState(1)
      // let counter = result[0];
      // let setCounter = result[1];
      // const [counter, setCounter] = result;
      
      // 使用 useState()
      const [counter, setCounter] = useState(1);    // 解构

      const addHandler = () => {
        // counter++        直接在这里修改counter不行，修改了但没重新渲染
        setCounter(counter + 1);    // 更改counter的值，将counter值++
      };

      const lessHandler = () => {
        setCounter(counter - 1);
      };

      return <div className={'app'}>
        <h1>{counter}</h1>
        <button onClick={lessHandler}>-</button>
        <button onClick={addHandler}>+</button>
      </div>;
    };
    export default App;

  • 注意：

    • 当state的值是一个对象时，修改时是使用新的对象去替换覆盖已有对象

      • 对象判断有没有修改比较的是内存地址而不是对象里面的属性和值，即使新旧对象一模一样也会重新渲染，因为地址不一样。如果新旧对象的地址一样，属性和值不一样，也不会重新渲染
      • 当通过setState去修改一个state（第一个参数）时，并不表示修改了第一个参数的值（初始值一开始是多少就是多少，不会改），它修改的是组件下一次渲染时state的值
      const [user, setUser] = useState({name: '孙悟空', age: 18});   // user就是{name: '孙悟空', age: 18}
      const updateUserHandler = () => {
          setUser({name:'猪八戒'});      // 把{name:'猪八戒'}替换{name: '孙悟空', age: 18}
          
          // 如果只想修改对象中某一个属性：
              // 直接修改旧的state对象，由于对象还是那个对象，没有变，所以不会生效
                  // user.name = '猪八戒';   
                      // 这里直接修改了user里原来的那个对象{name: '孙悟空', age: 18}，即使修改成功了，但是react进行对比的时候，对象进行对比的时候比的是地址，而不是对象里面的属性和值的不同，地址没变，所以不会重新渲染
                  // console.log(user);
                  // setUser(user);
          
              // 1、浅拷贝
                 const newUser = Object.assign({}, user);     // assign() 将use复制一份给{}
                      // console.log(newUser === user)  -->   false，长得一样，但是地址不同了
                  newUser.name = '猪八戒';   // 修改里面的值
                  setUser(newUser);       // 将新的对象覆盖旧的{name: '孙悟空', age: 18}对象
              // 2、浅拷贝展开
                  setUser({...user, name: '猪八戒'});    // 在对象{}中把对象user展开，相当于是浅拷贝给这个对象
      };
      <button onClick={updateUserHandler}>点击</button>

    • setState()会触发组件的重新渲染，它是异步的（修改state的值不会立刻渲染：index.js中的root.render(

      )）

      import {useState} from "react"
      import "./App.css"
      export default function App() {
          const [counter, setCounter] = useState(0)
          console.log("我被渲染了！")
          function changeCounter(){
              setCounter(1)       // 队列1（先进先出）
              setCounter(2)       // 队列2
              setCounter(3)       // 队列3
              setCounter(4)       // 队列4
              setCounter(5)       // 执行覆盖前面所有队列，执行渲染一次，而不是调用一次setCount就会渲染一次
          }
          return <div className="app">
              <h1>{counter}</h1>
              <button onClick={changeCounter}>多次修改</button>
          </div>
      }

      • 所以当调用setState()需要用旧的state的值时，一定要注意有可能出现计算错误的情况（两次快速点击拿到的值都是旧的队列中上一个state的值，两次拿到的值都是上一次队列的state的值，值相同，所以点击了两次实则只改了一次）

        • 为了避免这种情况，可以通过为setState()传递回调函数的形式来修改state值

          import {useState} from "react"
          import "./App.css"
          export default function App() {
              let[counter, setCounter] = useState(0)
              function changeCounter() {
                setTimeout(() => {
                    // setCounter(counter++)

                    // setState()中回调函数的返回值将会成为新的state值，回调函数执行时，React会将最新的state值作为参数传递
                      // setCounter((prevCounter) => {        取一个名字prevCounter形参，接收state的值
                      //     return prevCounter + 1;
                      // });
                      setCounter(preCount => preCount + 1)
                }, 1000)
              }
              return <div className="app">
                 <h1>{counter}</h1>
                 <button onClick={changeCounter}>点击</button>
              </div>
          }

获取DOM对象（useRef）​

  1. 引入

    import {useRef} from "React"

  2. 创建一个存储DOM对象的容器，使用 useRef( ) 钩子函数（usestate也是钩子函数）

    • 钩子函数的注意事项：
      1. 只能用于函数组件或自定义组件
      2. 只能直接在函数组件的根中调用（函数组件里面的如果还有函数，不能在里面调用，不能嵌套）
    const h1Ref = useRef()

  3. 将容器设置为想要获取DOM对象元素的ref属性

    <h1 ref={h1Ref}>...</h1>

  4. 获取对象

    h1Ref.current
    // h1Ref.current.innerText = "哈哈"

    • 注意：

      • useRef()返回值是一个普通的js对象：{current:undefined}，所以我们自己创建一个 const h1Ref = {current:undefined} 对象也可以代替useRef()

        • 区别：
          • 我们创建的对象，组件每次重新渲染都会创建一个新的对象
          • useRef()创建的对象可以确保每次渲染获取到的都是同一个对象
          • 当需要一个对象不会因为组件的重新渲染而改变时，就可以使用useRef()

state​

  • 类组件中state统一存储到了实例对象的state属性中，可以通过 this.state 来访问
    • 通过 this.setState( ) 对其进行修改，修改的本质是创建一个新的对象覆盖原来的对象
    • 函数组件中，响应函数直接以函数的形式定义在组件中，但是在类组件中，响应函数是以类的方法来定义，之前的属性都会保留（修改本质就是创建新的对象覆盖原来的对象，只修改一个覆盖并不会把state的里别的属性值给弄没掉），但是这仅限于直接存储于state中的属性（如果是state里写的一个属性是一个对象，则会全部覆盖，需要先展开原来的对象，再写要修改的属性）

获取DOM对象​

  1. 创建一个属性用来存储DOM对象：divRef = React.createRef( )
  2. 将需要的元素标签设为ref属性：ref={this.divRef}
// App.js：
import React, {Component} from 'react';
import './App.css';
import User from "./components/User";

class App extends Component {
    render() {
        return (
            <div className="app">
                <User name='猪八戒' age={28} gender={'男'}/>
            </div>
        );
    }
}
export default App;
// User.js：
import React, {Component} from 'react';

class User extends Component {
    // 创建属性存储DOM对象
    divRef = React.createRef();

    // 定义state并向其中添加属性，在这边定义的属性和方法都不需要const，因为是类
    state = {   // 类的属性
        count: 0,
        test: '哈哈',
        obj: {name: '孙悟空', age: 18}     // 如果是修改对象里的属性，只修改一个会把其他的弄没
    };
    
    // 点击修改state，函数都以箭头函数的形式定义（因为this）
    clickHandler = () => {      // 类的方法
        this.setState({count: 10});     // 如果上面的state设置了count和test，这里只修改count并不会把test弄没掉
        this.setState(prevState => {    
            return {    // 跟函数组件不一样，需要返回一个新的state对象覆盖原来的state对象，覆盖不会把state里别的值弄没
                count: prevState.count + 1
            }
        });
        
        // 修改对象
        this.setState({
            // 必须先把上一个this.state.obj展开，然后修改的name覆盖上去，这点跟函数组件一样
            obj:{...this.state.obj, name:'沙和尚'} 
        })
        
        // 获取DOM标签
        console.log(this.divRef.current)
    };

    render() {
        return (
            <div ref={this.divRef}>     // 获取DOM对象
                <h1>{this.state.count} --- {this.state.test}</h1>
                <h2>{this.state.obj.name} --- {this.state.obj.age}</h2>
                <button onClick={this.clickHandler}>点</button>  // 调用clickHandler要this.clickHandler，因为是方法
                <ul>
                    <li>姓名：{this.props.name}</li>
                    <li>年龄：{this.props.age}</li>
                    <li>性别：{this.props.gender}</li>
                </ul>
            </div>
        );
    }
}
export default User;

双向绑定​

  • 非受控组件：不由state控制
  • 受控组件：由state控制并双向绑定（表单）
将表单中的数据存储到state中，然后将state设置为表单项value值，当表单项发生变化，state会随之变化，反之，state发生变化，表单项也会跟着改变
这种操作我们就称为双向绑定，这样表单就变成了一个受控组件
import React from 'react'
import { useState } from 'react'
import Card from '../UI/Card/Card'
import "./LogsForm.css"

const LogsForm = (props) => {
    // let inputDate = ''
    // let inputDesc = ''
    // let inputTime = ''

    // 双向绑定数据
    const [inputDate, setinputDate] = useState('')
    const [inputDesc, setinputDesc] = useState('')
    const [inputTime, setinputTime] = useState('')
    // 将表单数据统一到一个state中
    // const [formData, setFormData] = useState({
    //     inputDate: '',
    //     inputDesc: '',
    //     inputTime: ''
    // })

    const dateChangeHandler = (e) => {
        // inputDate = e.target.value
        setinputDate(e.target.value)
        // setFormData({
        //     // 如果只写inputDate: e.target.value不行，因为setFormData会覆盖上面原来的useState对象，所以要先进行浅复制-->展开，表示只替换当前的target
        //     ...formData,
        //     inputDate: e.target.value
        // })
    }

    const descChangeHandler = (e) => {
        // inputDesc = e.target.value
        setinputDesc(e.target.value)
        // setFormData({
        //     ...formData,
        //     inputDesc: e.target.value
        // })
    }

    const timeChangeHandler = (e) => {
        // inputTime = e.target.value
        setinputTime(e.target.value)
        // setFormData({
        //     ...formData,
        //     inputTime: e.target.value
        // })
    }

    // 监听form表单的submit事件
    const formSubmitHandler = (e) => {
        // 取消表单的submit默认跳转行为
        e.preventDefault();

        // 把数据封装成一个对象 
        const newLog = {
            date: new Date(inputDate),
            desc: inputDesc,
            time: +inputTime,
            // date: new Date(formData.inputDate),
            // desc: formData.inputDesc,
            // time: +formData.inputTime,
        }

        // 当要添加新日志时，调用App.js传过来的onSaveLog函数，然后把数据newLog传给这个函数，父组件这个函数形参接收
        props.onSaveLog(newLog)

        // 清空state的内容，setState默认重新渲染数据，这里渲染成空字符串
        setinputDate('')
        setinputDesc('')
        setinputTime('')
        // setFormData({
        //     inputDate: '',
        //     inputDesc: '',
        //     inputTime: ''
        // })
    }

    return (
        <Card className="logs-form">
            <form onSubmit={formSubmitHandler}>
                {/* onchange事件用来监听内容的变化，内容一变化就自动调用事件 */}
                <div className='form-item'>
                    <label htmlFor="date">日期</label>
                    {/* 这里加value={inputDate}的目的是为了提交表单后清空input的内容，因为上面获取完数据提交表单后setinputDate('')了，setState后默认会重新渲染数据，上面重新渲染了，这里就清空了 */}
                    <input onChange={dateChangeHandler} value={inputDate} id='date' type="date" />
                    {/* <input onChange={dateChangeHandler} value={formData.inputDate} id='date' type="date" /> */}
                </div>
                <div className='form-item'>
                    <label htmlFor="desc">内容</label>
                    <input onChange={descChangeHandler} value={inputDesc} id='desc' type="text" />
                    {/* <input onChange={descChangeHandler} value={formData.inputDesc} id='desc' type="text" /> */}
                </div>
                <div className='form-item'>
                    <label htmlFor="time">时长</label>
                    <input onChange={timeChangeHandler} value={inputTime} id='time' type="number" />
                    {/* <input onChange={timeChangeHandler} value={formData.inputTime} id='time' type="number" /> */}
                </div>
                <div className='form-btn'>
                    <button>添加</button>
                </div>
            </form>
        </Card>
    )
}

export default LogsForm

import Logs from "./Components/Logs/Logs";

import "./App.css"
import { useState } from "react";

const App = () => {

    const [logsData, setLogsData] = useState([
        {

            time: 30
        },
        {
            id: '002',
            date: new Date(2022, 2, 10, 12, 30),
            desc: '学习降龙十八掌',
            time: 20
        },
        {
            id: '003',
            date: new Date(2022, 2, 11, 11, 30),
            desc: '学习JavaScript',
            time: 40
        },
        {
            id: '004',
            date: new Date(2022, 2, 15, 10, 30),
            desc: '学习React',
            time: 80
        }
    ])

    // 定义一个函数，形参接收从子组件调用这个函数的数据
    const saveLogHandler = (newLog) => {
        // 为新的日志生成一个id
        newLog.id = "00" + (logsData.length + 1)

        // logsData.push(newLog)

        setLogsData([...logsData, newLog])
    }

    return <div className="app">

        <LogsForm onSaveLog={saveLogHandler}></LogsForm>
        <Logs logsData={logsData} />
    </div>
};

Context​

  • 目前传值是通过props一层一层传，如果是要给隔了很远的兄弟组件传则非常的麻烦
  • Context是在外层组件统一设置一些公共数据（公共存储空间），在内层的所有的组件都可以访问到Context中的数据，相当于全局作用域
  • Context流程：先把数据传给外部的Context，然后在组件里使用Context读取数据，不用再一层一层传了

  1. 创建Context（src下创建store文件夹，创建XXXContext.js）

    import React from "react";

    // 创建Context
    const TestContext = React.createContext({
        // 注意：数据信息并不能直接储存在Context里，即使把数据存在这里，因为没有state，改了这边的数据也不会发生重新渲染，写在这边的数据可以作为默认数据
        name: '孙悟空',
        age: 18
    })

    export default TestContext;

  2. 使用Context

    • 第一种方式：采用消费者Consumer模式

      // 引入Context
      import TestContext from '../store/textContext'

      const A = () => {
          return (
              // 创建TestContext.Consumer标签
              <TestContext.Consumer>
                  {
                      // 参数为函数，定义实参接收数据
                      ctx => {
                          return (
                              <div>
                                  {ctx.name}-{ctx.age}
                              </div>
                          )
                      }
                  }
              </TestContext.Consumer>
          )
      }

      export default A

    • 第二种方式：采用钩子函数（只能使用于函数组件）

      // 引入钩子函数useContext
      import { useContext } from 'react'
      // 引入Context
      import TestContext from '../store/textContext'

      const A = () => {
          // 使用钩子函数获取Context
          const ctx = useContext(TestContext)

          return (
              <div>
                  {ctx.name}-{ctx.age}
              </div>
          )
      }

      export default A
